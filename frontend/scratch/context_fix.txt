  /** อนุมัติหรือปฏิเสธสมาชิก (status: approved / rejected) */
  const updateMemberStatus = useCallback(
    async (groupId, userId, status) => {
      const url = `/api/groups/${groupId}/members/${userId}`;
      return apiFetch(url, {
        method: 'PATCH',
        headers: authHeaders(token),
        body: JSON.stringify({ join_status: status }),
      });
    },
    [token]
  );

  /** ลบกลุ่ม (เฉพาะเจ้าของ) */
  const deleteGroup = useCallback(
    async (groupId) => {
      return apiFetch(`/api/groups/${groupId}`, {
        method: 'DELETE',
        headers: authHeaders(token),
      });
    },
    [token]
  );

  /** ออกจากกลุ่ม */
  const leaveGroup = useCallback(
    async (groupId) => {
      return apiFetch(`/api/groups/${groupId}/leave`, {
        method: 'POST',
        headers: authHeaders(token),
      });
    },
    [token]
  );
